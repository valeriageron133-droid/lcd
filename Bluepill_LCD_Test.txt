/** ###################################################################
** @brief
**    Evento       : TimerInt_OnInterrupt()
**    Proyecto     : STM32F103C8_Test_LCD
**    Procesador   : STM32F103C8T6
**    Herramienta  : Mbed
**    Compilador   : Mbed Online C Compiler
**    Version      : Mbed-OS 5.15.0
**    Compiler     : GNU C Compiler
**    Date/Time    : 18-12-2019, 11:48, # CodeGen: 0
**    Abstract     :
**         This is user's event module.
**         Put your event handler code here.
**    Settings    :
**    Contents    :
**         Cpu_OnNMIINT - void Cpu_OnNMIINT(void);
**
** ###################################################################*/
void TimerInt_OnInterrupt()           // Rutina de Atención al Ticker
{
    counter--;                          // Aquí va la Rutina de Servicio !
    if (!counter) 
    {
        N_Interrupciones++;             // Incrementa el contador 
        terminal.printf("Counter Finish! %d Times \r",N_Interrupciones);
        led_monitor = !led_monitor;     // Parapadeo del LED por Interrupción (Toggle the LED)
        if (!led_monitor)heartbeat++;   // Incrementa el contador de Parpadeos
        counter = Rate;                 // Restablece el contador
    }
}

void Setup()

*********************************************************************/

    terminal.printf("\e[2J\e[H");   // Clear Screen

*********************************************************************/
    terminal.printf("Testing Buzzer... \n\r");

*   Configufración de la Interfaz al LCD                            *
*********************************************************************/
    Init_Interfaces();
/********************************************************************
*   Prueba de instalación de la interface al LCD    Mucho OjO !!!   *
*********************************************************************/

/*--------------------------------------------------------------------
|   Configuración de la dirección de datos en la interface I2C
---------------------------------------------------------------------- */
#if LCD_INTERFACE == 1
    terminal.printf("LCD on I2C interface Address: \e[1;33;40m0x%X \e[0m\n\r",I2C_ADDRESS);  
#endif
/*--------------------------------------------------------------------
|   Configuración de la dirección de datos en la interface SPI
---------------------------------------------------------------------- */
#if LCD_INTERFACE == 2
    terminal.printf("LCD on SPI interface \n\r"); 
    if ((SPI1 -> CR1 & 0x0080) > 0) terminal.printf ("SPI LSB First \n\r");
    else terminal.printf ("SPI MSB First \n\r");

---------------------------------------------------------------------- */

/********************************************************************

void Init_Interfaces() 
{
/********************************************************************
*   Configuración del Bus I2C      / Mucho OjO !!!                  *
*********************************************************************/
#if LCD_INTERFACE == 1           /* LCD Utilizando el I2C */
/*--------------------------------------------------------------------
|   Configuración utilizando las Classes de Mbed 
---------------------------------------------------------------------- */
/* 
    i2c_wire.frequency(1000000);    // Frecuencia de operación para el I2C (1MHz)

*/
/*--------------------------------------------------------------------
|   Configufración del Bus I2C utilizando HAL y los Registros 
|   Configuración utilizando los Registros (Consultar RM0008 pag. 758)
---------------------------------------------------------------------- */
//    HAL_RCC_APB2ENR               // Registros de Configuración para Habilitación de Reloj de Periféricos
RCC -> APB1ENR |= 0x02000000;       // Habilita el I2C1 (I2C1_EN)
RCC -> APB2ENR |= 0x00000001;       // Habilita las Funciones Avanzadas de E/S (AFIO_EN)
//    HAL_I2C_CR1                   // Registros de Control del I2C1
I2C1 -> CR1 |= 0x0400;              // CK: Acknowledge enable
I2C1 -> CR1 |= 0x0200;              // STOP: Stop generation
I2C1 -> CR1 |= 0x0100;              // START: Start generation
I2C1 -> CR1 |= 0x0020;              // ENPEC: PEC enable
I2C1 -> CR1 |= 0x0002;              // SMBUS: SMBus mode
I2C1 -> CR1 |= 0x0001;              // PE: Peripheral enable */
    terminal.printf("I2C Initialized...\n\r");
/*--------------------------------------------------------------------
|   Busca Dispositivos I2C conectados al Bus
---------------------------------------------------------------------- */
    I2C_Scanner();                  // Busca Dispositivos I2C conectados al Bus
#endif

/********************************************************************
*   Configufración del SPI          / Mucho OjO !!!                 *
*********************************************************************/
#if LCD_INTERFACE == 2    /* LCD Utilizando el SPI */
/*--------------------------------------------------------------------
|   Prueba del SN74HC595 a la Interface al SPI 
---------------------------------------------------------------------- */
    terminal.printf("\rPrueba del SPI y SN74HC595 \r\n ");
    SSel=FALSE;                             // Activa la señal SS
    int respuesta = spi_lcd.write(0xFF);
    SSel=TRUE;                              // Desactiva la señal SS
    SSel=FALSE;                             // Activa la señal SS
        respuesta = spi_lcd.write(0xAA);
    SSel=TRUE;                              // Desactiva la señal SS
    SSel=FALSE;                             // Activa la señal SS
        respuesta = spi_lcd.write(0x55);
    SSel=TRUE;                              // Desactiva la señal SS
    SSel=FALSE;                             // Activa la señal SS
        respuesta = spi_lcd.write(0x00);
    SSel=TRUE;                              // Desactiva la señal SS
    terminal.printf("\rSPI funciona OK! \r\n");

/*--------------------------------------------------------------------
|   Configuración utilizando las Classes de Mbed 
---------------------------------------------------------------------- */
    spi_lcd.frequency(1000000);      // Frecuencia de operación para el SPI 1MHz
    spi_lcd.format(8,0);            // Número de Bits y Modo de Operación para el SPI POL=1 PHA=1
//    spi_lcd.frame(1);             // Formato de la Trama para el SPI
    SPI1 -> CR1 |= 0x0080;          // Formato de la Trama (LSBFIRST): 0 = MSB (default), 1= LSB
    SSel=FALSE;                     // Desactiva el Latch del SN74HC595  
    terminal.printf("SPI Initialzed...\n\r");
#endif
}

/*! @brief
** ===================================================================
**    Función     :     Función I2C_Scanner()   
**    Descripción :     Busca dispositivos conectados al Bus I2C
**    Parámetros  :     Ninguno        
**    Retorna     :     Nada
** ===================================================================
*/
void I2C_Scanner()
{
#if LCD_TYPE == 1                       // Definición para pines del LCD con interfaz I2C
    terminal.printf ("I2C scanner. Scanning ...");
    uint8_t count = 0;
    char     data;
    char     test[] = {0x55,0xAA,0x0F,0xF0,0xFF,0x00};      // Patrón de Prueba
    i2c_lcd.frequency(100000);          // Frecuencia de operación para el I2C (100KHz)
    i2c_lcd.start();                    // Inicio de operación para el I2C  
    for (int i = 8; i < 128; i++) {
        uint8_t acknoledge = i2c_lcd.write (i,test,6,1);    //(dirección, ap_datos, N°_datos, repetición)
//    uint8_t acknoledge = i2c_lcd.write (i,test,6);          //(dirección, ap_datos, N°_datos, repetición)
        if (acknoledge == 0) {
            terminal.printf ("\n\rFound address: ");
            terminal.printf ("%u DEC",i);                   // Valor de la dirección en Decimal
            terminal.printf (" (0x%x HEX)\n\r",i);          // Valor de la dirección en Hexadecimal
            data = i2c_lcd.read (i);    //(dirección, ap_datos, N°_datos, repetición)
            terminal.printf ("Data: (0x%hd HEX)\n\r",data); // Valor de la dirección en Hexadecimal
            count++;
            wait (1);                   // maybe unneeded?
        } // end of good response
        led_monitor = !led_monitor;     // Parapadeo del LED
    } // end of for loop
    led_monitor = 1;                    // Apaga el LED
    i2c_lcd.stop();                     // Fin de operación para el I2C
    terminal.printf ("\n\rDone.\n\r");
    terminal.printf ("Found %d Device(s)\n\r",count);
#endif
}  // end of I2C Scanner

        lcd.putc('0' + row);                // Enumera los renglones

            lcd.putc('*');                  // Llena la línea con el caracter "*"

        }
        terminal.printf("MemAddr(Col=%d, Row=%d)=0x%02X\n\r", col, row, lcd.getAddress(col, row));

        lcd.putc('+');                      // Al final de la línea agrega el "+"

    }
    Elpased_Time.stop();                    // Detiene el Contador de Ticks
    terminal.printf("All my hard work took %f sec\r\n", Elpased_Time.read()); // Calcula el tiempo transcurrido
    wait(1);                                // Espera por 1 segundo
    terminal.printf("LCD Display and TextLCD Enhanced Test Completed.\n\r");
    wait(1);                                // Espera 1 segundo
    lcd.cls();                              // Limpia la Pantalla
//  lcd.setBacklight(TextLCD::LightOff);      // Apaga la Iluminación de Fondo
    lcd.setCursor(TextLCD::CurOff_BlkOn);   // Apaga el Cursor y Enciende el Parpadeo
// Set and show user defined characters. A maximum of 8 UDCs are supported by the HD44780.
// They are defined by a 5x7 bitpattern.
    lcd.setUDC(0, (char *) udc_0);          // Selecciona el caracter para el cursor |>
    lcd.putc(0);                            // Lo imprime en la Pantalla LCD
    lcd.printf(" LCD Ready... ");           // El LCD está listo para usarse
    wait(2);                                // Espera 2 segundos
    lcd.setCursor(TextLCD::CurOff_BlkOff);  // Apaga el Cursor y Apaga el Parpadeo
    terminal.printf("LCD Starting OK! ... \n\r");
}

*+   Para escribir un Byte en Modo de 4 bits en el LCD el procedimiento consiste 
*+   en escribir primeramente el Nibble Más Significativo (MSN) seguido del 
*+   Nibble Menos Significativo (LSN).
*+
*+   El procedimeieto para enviar un comando o escribir un caracter en el display
*+   consiste de 4 escrituras al Display LCD y es como sigue:
*+   1.- Seleccionar en la señal RS (Register/String) el comando o dato a escribir
*+        RS = 0  Instrucción   (Registro)
*+        RS = 1  Dato          (Cadena)
*+   2.- Seleccionar en la señal E  (Enable) en 0 = Deshabilitada la 
*+       Lectura/Escritura al Display LCD
*+   3.- Seleccionar en la señal RW (Read/Write) en 0 = Escritura
*+   4.- Seleccionar el Nibble a escrbir (MSN) o (LSN)
*+   5.- Enviar las señales anteriores a los pines del Display LCD
*+   6.- Cambiar la señal E (Enable) a 1 para havilitar la escritura al LCD
*+   7.- Enviar las señales anteriores a los pines del Display LCD

*+      X   0   1   1    X    X    X    X       Escribe Nibble Más Significativo
*+      X   0   0   1    x    x    x    x       Prepara Nibble Menos Significativo
*+      X   0   1   1    x    x    x    x       Escribe Nibble Menos Significativo

*+        Para enviar cualquier instrucción o dato (RS) hay que hacer una    __
